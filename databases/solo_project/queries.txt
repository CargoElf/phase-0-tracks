**DB SETUP**

CREATE TABLE good_alignment (
  id INTEGER PRIMARY KEY,
  good BOOLEAN
);

CREATE TABLE characters (
  id INTEGER PRIMARY KEY,
  name VARCHAR(255),
  level INTEGER,
  good BOOLEAN,
  FOREIGN KEY(good) REFERENCES good_alignment(id)
);
