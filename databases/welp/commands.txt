CREATE TABLE business (
  id INTEGER PRIMARY KEY,
  name varchar(255)
);

CREATE TABLE users (
  id INTEGER PRIMARY KEY,
  first_name varchar(255),
  last_name varchar(255)
);

CREATE TABLE reviews (
  id INTEGER PRIMARY KEY,
  stars INT,
  comment varchar(255),
  business_id INT,
  user_id INT,
  FOREIGN KEY (business_id) REFERENCES business(id),
  FOREIGN KEY (user_id) REFERENCES users(id)
);
